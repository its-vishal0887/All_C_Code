#include<stdio.h>
#include<stdlib.h>
#define MAX 5

int queue[MAX];
int front = -1, rear = -1;

void enqueue(int order){
    if(rear == MAX - 1){
        printf("\n Queue OverFlow! no more orders can be accepted.\n");
    }else{
        if(front == -1){
            front = 0;
        }
        rear++;
        queue[rear] = order;
        printf("\nOrder #%d Added Successfully!\n", order);
    }
}

void dequeue(){
    if(front == -1){
        printf("\nNo order is pending.\n");
    }else{
        printf("\nOrder #%d served successfully!\n", queue[front]);
        front++;
        if(front > rear){
            front = rear -1;
        }
    }
}

void display(){
    if(front == -1){
        printf("\nNo Pending Orders.\n");
    }else{
        printf("\n--- Current Orders in queue---\n");
        for(int i = front; i<=rear; i++){
            printf("Order #%d\n", queue[i]);
        }
    }
}

int main(){
    int choice, orderNum = 1;

    do{
        printf("\n===Pizza Parlor Job Queue===\n");
        printf("1. Add New Order (Enqueue)\n");
        printf("2. Serve Order (Dequeue)\n");
        printf("3. Display Orders \n");
        printf("0.Exit\n");
        printf("Enter Your choice: ");
        scanf("%d", &choice);

        switch(choice){
            case 1: 
                enqueue(orderNum++);
                break;
            
            case 2:
                dequeue();
                break;

            case 3:
                display();
                break;

            case 0:
                printf("\nExiting...Thank You for visiting the pizza parlor!\n");
                break;

            default:
                printf("\nInvalid choice ! try again");
        }
    
    }while(choice != 0);
    return 0;
}
